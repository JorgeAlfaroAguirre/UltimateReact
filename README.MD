# Resumen de react

```typescript
import { useEffect, useState } from "react";

function App() {
  const [count, setCount] = useState(0);
  useEffect(() => {
    console.log(count);
  }, [count]); // Se ejecuta cada vez que count cambia
  const handleClick = () => {
    setCount((prevCount) => prevCount + 1); // Usar el valor previo para evitar problemas con estados asincr√≥nicos
  };

  return (
    <>
      <button onClick={handleClick}>+1</button>
      {count}
    </>
  );
}

export default App;
```
